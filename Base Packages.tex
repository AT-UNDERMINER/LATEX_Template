% Essential Packages  
\usepackage{graphicx}   % For including images (resizing, rotating, and placing images in the document)  
\usepackage{amsmath}    % Advanced mathematical typesetting (supports complex equations and alignment)  
\usepackage{amssymb}    % Additional mathematical symbols (extends symbol collection for math mode)  
\usepackage{amsthm}     % Theorem environments (provides easy formatting for theorems, lemmas, etc.)  

% Formatting and Layout  
\usepackage{titlesec}   % Customize section titles (adjust font size, spacing, and numbering)  
\usepackage{multicol}   % Multi-column layout (allows text to be displayed in multiple columns)  
\usepackage{parskip}    % Adjust paragraph spacing (removes indentation and adds spacing between paragraphs)  

% Figures and Captions  
\usepackage{caption}    % Customizing captions (allows formatting figure and table captions)  
\usepackage{subcaption} % Subfigures (enables multiple figures under a single caption)  
\usepackage{float}      % Precise figure placement (provides control using [H] option for strict placement)  

% Colors and Text  
\usepackage{xcolor}     % Colored text (define and use colors in text and backgrounds)  
\usepackage{textcomp}   % Extra symbols (provides additional text symbols like degree °)  
\usepackage{gensymb}    % Additional symbols (including ° for degrees)  

% References and Citations  
\usepackage{hyperref}   % Hyperlinks (creates clickable references, URLs, and cross-references)  
\usepackage{cite}       % Citation management (ensures proper formatting and sorting of citations)  

% Code and Listings  
\usepackage{listings}   % Code inclusion (formats and highlights source code snippets)  

% SI Units and Measurements  
\usepackage{siunitx}    % Proper typesetting of units and numbers (handles SI units consistently)  

% Drawing and Diagrams  
\usepackage{tikz}       % General drawing tool (allows creating diagrams, graphs, and free-body diagrams)  
\usepackage{circuitikz} % Electrical circuit diagrams (extends TikZ for electrical circuit drawings)  
\usepackage{pgfplots}   % Plots and graphs (creates high-quality function and data plots)  
\pgfplotsset{compat=newest} % Ensure the latest PGFPlots features are enabled  

% Miscellaneous  
\usepackage{lipsum}     % Placeholder text (generates dummy text for testing layout)  
\usepackage{layout}     % Visualize document layout (displays margins and spacing)  
\usepackage{mwe}        % Placeholder images and text (provides sample content for testing)
